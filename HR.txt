        target.HIRE_DATE = source.HIRE_DATE,
        target.JOB_ID = source.JOB_ID,
        target.LAST_NAME = source.LAST_NAME,
        target.MANAGER_ID = source.MANAGER_ID,
        target.PHONE_NUMBER = source.PHONE_NUMBER,
        target.SALARY = source.SALARY
WHEN NOT MATCHED THEN
    INSERT (EMPLOYEE_ID, ADDRESS, DEPARTMENT_ID, EMAIL, FIRST_NAME, HIRE_DATE, JOB_ID, LAST_NAME, MANAGER_ID, PHONE_NUMBER, SALARY)
    VALUES (source.EMPLOYEE_ID, source.ADDRESS, source.DEPARTMENT_ID, source.EMAIL, source.FIRST_NAME, source.HIRE_DATE, source.JOB_ID, source.LAST_NAME, source.MANAGER_ID, source.PHONE_NUMBER, source.SALARY);


    -------------------------
    
    string functions 

    
    SELECT 
  FIRST_NAME,
  LAST_NAME,
  UPPER(FIRST_NAME) as UPPER_FIRST_NAME,
  LOWER(LAST_NAME) as LOWER_LAST_NAME,
  LENGTH(FIRST_NAME) as NAME_LENGTH,
  CONCAT(FIRST_NAME, ' ', LAST_NAME) as FULL_NAME,
  LEFT(FIRST_NAME, 3) as FIRST_THREE_CHARS,
  RIGHT(LAST_NAME, 3) as LAST_THREE_CHARS,
  TRIM(ADDRESS) as TRIMMED_ADDRESS,
  REPLACE(EMAIL, '@', '_AT_') as MODIFIED_EMAIL,
  SUBSTRING(FIRST_NAME, 1, 4) as FIRST_FOUR_CHARS,
  INITCAP(LOWER(FIRST_NAME)) as PROPER_CASE_NAME,
  LTRIM(RTRIM(ADDRESS)) as TRIMMED_BOTH_SIDES,
  POSITION(' ' IN CONCAT(FIRST_NAME, ' ', LAST_NAME)) as SPACE_POSITION
FROM HR.PUBLIC.EMPLOYEES;

    Number Functions

    
SELECT 
  EMPLOYEE_ID,
  FIRST_NAME,
  LAST_NAME,
  SALARY,
  ABS(SALARY) as absolute_salary,
  CEIL(SALARY) as ceiling_salary,
  FLOOR(SALARY) as floor_salary,
  ROUND(SALARY, 2) as rounded_salary,
  POWER(SALARY, 2) as salary_squared,
  SQRT(SALARY) as salary_square_root,
  MOD(SALARY, 1000) as salary_modulus,
  SIGN(SALARY) as salary_sign
FROM HR.PUBLIC.EMPLOYEES;


----------------------------------------------------------------------
     Date Functions   
     

SELECT 
  EMPLOYEE_ID,
  FIRST_NAME,
  LAST_NAME,
  DATE_TRUNC('MONTH', HIRE_DATE) AS MONTH_START,
  LAST_DAY(HIRE_DATE) AS MONTH_END,
  DATEADD('YEAR', 1, HIRE_DATE) AS ONE_YEAR_LATER,
  DATEDIFF('DAY', HIRE_DATE, CURRENT_DATE()) AS DAYS_EMPLOYED,
  DATEDIFF('MONTH', HIRE_DATE, CURRENT_DATE()) AS MONTHS_EMPLOYED,
  DATEDIFF('YEAR', HIRE_DATE, CURRENT_DATE()) AS YEARS_EMPLOYED,
  TO_VARCHAR(HIRE_DATE, 'YYYY-MM-DD') AS FORMATTED_DATE,
  DAYNAME(HIRE_DATE) AS HIRE_DAY_NAME,
  MONTHNAME(HIRE_DATE) AS HIRE_MONTH_NAME,
  YEAR(HIRE_DATE) AS HIRE_YEAR,
  QUARTER(HIRE_DATE) AS HIRE_QUARTER
FROM HR.PUBLIC.EMPLOYEES;

SELECT 
  COUNT(*) as employee_count,
  COUNT(DISTINCT department_id) as unique_departments,
  AVG(salary) as avg_salary,
  MIN(salary) as min_salary,
  MAX(salary) as max_salary,
  SUM(salary) as total_salary_cost,
  MEDIAN(salary) as median_salary,
  STDDEV(salary) as salary_stddev,
  COUNT(manager_id) as employees_with_managers,
  MIN(hire_date) as earliest_hire_date,
  MAX(hire_date) as latest_hire_date
FROM HR.PUBLIC.EMPLOYEES;
