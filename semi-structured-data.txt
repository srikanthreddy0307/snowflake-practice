------------ Load data using INFER_SCHEMA (no manual table creation)

CREATE OR REPLACE STAGE MANAGE_DB.EXTERNAL_STAGES.CSVSTAGE
    url='s3://bucketsnowflakes3';

LIST @MANAGE_DB.EXTERNAL_STAGES.CSVSTAGE;

CREATE OR REPLACE FILE FORMAT MANAGE_DB.EXTERNAL_STAGES.MY_CSV_FORMAT
    TYPE = csv
    PARSE_HEADER = true;

------------ 1) SAMPLEDATA (sampledata.csv - 2 columns: AGE, CUSTOMERNAME)

SELECT * FROM TABLE(
    INFER_SCHEMA(
        LOCATION => '@MANAGE_DB.EXTERNAL_STAGES.CSVSTAGE/sampledata.csv',
        FILE_FORMAT => 'MANAGE_DB.EXTERNAL_STAGES.MY_CSV_FORMAT'
    )
);

CREATE OR REPLACE TABLE MANAGE_DB.EXTERNAL_STAGES.SAMPLEDATA
    USING TEMPLATE (
        SELECT ARRAY_AGG(OBJECT_CONSTRUCT(*))
        FROM TABLE(
            INFER_SCHEMA(
                LOCATION => '@MANAGE_DB.EXTERNAL_STAGES.CSVSTAGE/sampledata.csv',
                FILE_FORMAT => 'MANAGE_DB.EXTERNAL_STAGES.MY_CSV_FORMAT'
            )
        )
    );

COPY INTO MANAGE_DB.EXTERNAL_STAGES.SAMPLEDATA
    FROM @MANAGE_DB.EXTERNAL_STAGES.CSVSTAGE/sampledata.csv
    FILE_FORMAT = (FORMAT_NAME = 'MANAGE_DB.EXTERNAL_STAGES.MY_CSV_FORMAT')
    MATCH_BY_COLUMN_NAME = CASE_INSENSITIVE;

SELECT * FROM MANAGE_DB.EXTERNAL_STAGES.SAMPLEDATA;

------------ 2) ORDERDETAILS (OrderDetails.csv - 6 columns)
