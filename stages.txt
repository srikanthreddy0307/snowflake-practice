--- Separate Profit and Loss with Status

SELECT
    ORDER_ID,
    AMOUNT,
    PROFIT,
    QUANTITY,
    CATEGORY,
    SUBCATEGORY,
    CASE WHEN PROFIT >= 0 THEN PROFIT ELSE 0 END AS PROFIT,
    CASE WHEN PROFIT < 0 THEN ABS(PROFIT) ELSE 0 END AS LOSS,
    CASE WHEN PROFIT >= 0 THEN 'Profit' ELSE 'Loss' END AS STATUS
    
FROM OUR_FIRST_DB.PUBLIC.ORDERS; 

--- Total Profit and Total Loss
SELECT
    SUM(CASE WHEN PROFIT >= 0 THEN PROFIT ELSE 0 END) AS TOTAL_PROFIT,
    SUM(CASE WHEN PROFIT < 0 THEN ABS(PROFIT) ELSE 0 END) AS TOTAL_LOSS
FROM OUR_FIRST_DB.PUBLIC.ORDERS;
