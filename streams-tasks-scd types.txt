-- =============================================================
-- SCD TYPE 1: Overwrite (no history, latest values only)
-- =============================================================

CREATE OR REPLACE TABLE VITECH_DEV.SILVER.ORDERS_SCD1 (
    ORDER_ID NUMBER PRIMARY KEY,
    CUSTOMER_ID NUMBER NOT NULL,
    ORDER_DATE DATE NOT NULL,
    ORDER_STATUS VARCHAR(50) NOT NULL,
    TOTAL_AMOUNT NUMBER(12,2) NOT NULL,
    SHIPPING_ADDRESS VARCHAR(500),
    LAST_UPDATED TIMESTAMP_LTZ DEFAULT CURRENT_TIMESTAMP()
);

CREATE OR REPLACE STREAM VITECH_DEV.BRONZE.ORDERS_STREAM_SCD1
    ON TABLE VITECH_DEV.BRONZE.ORDERS APPEND_ONLY = FALSE;

CREATE OR REPLACE TASK VITECH_DEV.SILVER.ORDERS_TASK_SCD1
    WAREHOUSE = COMPUTE_WH
    SCHEDULE = 'USING CRON * * * * * UTC'
    WHEN SYSTEM$STREAM_HAS_DATA('VITECH_DEV.BRONZE.ORDERS_STREAM_SCD1')
    AS
MERGE INTO VITECH_DEV.SILVER.ORDERS_SCD1 AS TGT
USING VITECH_DEV.BRONZE.ORDERS_STREAM_SCD1 AS SRC
    ON TGT.ORDER_ID = SRC.ORDER_ID
WHEN MATCHED AND SRC.METADATA$ACTION = 'DELETE' AND SRC.METADATA$ISUPDATE = FALSE THEN
    DELETE
WHEN MATCHED AND SRC.METADATA$ACTION = 'INSERT' THEN
    UPDATE SET
        TGT.CUSTOMER_ID = SRC.CUSTOMER_ID,
        TGT.ORDER_DATE = SRC.ORDER_DATE,
        TGT.ORDER_STATUS = SRC.ORDER_STATUS,
        TGT.TOTAL_AMOUNT = SRC.TOTAL_AMOUNT,
        TGT.SHIPPING_ADDRESS = SRC.SHIPPING_ADDRESS,
        TGT.LAST_UPDATED = CURRENT_TIMESTAMP()
WHEN NOT MATCHED AND SRC.METADATA$ACTION = 'INSERT' THEN
    INSERT (ORDER_ID, CUSTOMER_ID, ORDER_DATE, ORDER_STATUS, TOTAL_AMOUNT, SHIPPING_ADDRESS, LAST_UPDATED)
    VALUES (SRC.ORDER_ID, SRC.CUSTOMER_ID, SRC.ORDER_DATE, SRC.ORDER_STATUS, SRC.TOTAL_AMOUNT, SRC.SHIPPING_ADDRESS, CURRENT_TIMESTAMP());

ALTER TASK VITECH_DEV.SILVER.ORDERS_TASK_SCD1 RESUME;

select * from VITECH_DEV.BRONZE.ORDERS_STREAM_SCD1;
select * from VITECH_DEV.SILVER.ORDERS_SCD1;


-- =============================================================
-- SCD TYPE 2: Full history tracking (new row per change)
-- =============================================================

CREATE OR REPLACE TABLE VITECH_DEV.SILVER.ORDERS_SCD2 (
    ORDER_KEY NUMBER AUTOINCREMENT PRIMARY KEY,
    ORDER_ID NUMBER NOT NULL,
    CUSTOMER_ID NUMBER NOT NULL,
    ORDER_DATE DATE NOT NULL,
    ORDER_STATUS VARCHAR(50) NOT NULL,
    TOTAL_AMOUNT NUMBER(12,2) NOT NULL,
    SHIPPING_ADDRESS VARCHAR(500),
    EFFECTIVE_FROM TIMESTAMP_LTZ DEFAULT CURRENT_TIMESTAMP(),
    EFFECTIVE_TO TIMESTAMP_LTZ DEFAULT '9999-12-31'::TIMESTAMP_LTZ,
    IS_CURRENT BOOLEAN DEFAULT TRUE
);

CREATE OR REPLACE STREAM VITECH_DEV.BRONZE.ORDERS_STREAM_SCD2
    ON TABLE VITECH_DEV.BRONZE.ORDERS APPEND_ONLY = FALSE;

CREATE OR REPLACE TASK VITECH_DEV.SILVER.ORDERS_TASK_SCD2
    WAREHOUSE = COMPUTE_WH
    SCHEDULE = 'USING CRON * * * * * UTC'
    WHEN SYSTEM$STREAM_HAS_DATA('VITECH_DEV.BRONZE.ORDERS_STREAM_SCD2')
    AS
MERGE INTO VITECH_DEV.SILVER.ORDERS_SCD2 AS TGT
USING (
    SELECT ORDER_ID, CUSTOMER_ID, ORDER_DATE, ORDER_STATUS, TOTAL_AMOUNT, SHIPPING_ADDRESS,
           METADATA$ACTION, METADATA$ISUPDATE
    FROM VITECH_DEV.BRONZE.ORDERS_STREAM_SCD2
) AS SRC
    ON TGT.ORDER_ID = SRC.ORDER_ID AND TGT.IS_CURRENT = TRUE
WHEN MATCHED AND SRC.METADATA$ACTION = 'DELETE' AND SRC.METADATA$ISUPDATE = FALSE THEN
    UPDATE SET
        TGT.IS_CURRENT = FALSE,
        TGT.EFFECTIVE_TO = CURRENT_TIMESTAMP()
WHEN MATCHED AND SRC.METADATA$ACTION = 'INSERT' THEN
    UPDATE SET
        TGT.IS_CURRENT = FALSE,
        TGT.EFFECTIVE_TO = CURRENT_TIMESTAMP()
WHEN NOT MATCHED AND SRC.METADATA$ACTION = 'INSERT' THEN
    INSERT (ORDER_ID, CUSTOMER_ID, ORDER_DATE, ORDER_STATUS, TOTAL_AMOUNT, SHIPPING_ADDRESS, EFFECTIVE_FROM, EFFECTIVE_TO, IS_CURRENT)
    VALUES (SRC.ORDER_ID, SRC.CUSTOMER_ID, SRC.ORDER_DATE, SRC.ORDER_STATUS, SRC.TOTAL_AMOUNT, SRC.SHIPPING_ADDRESS, CURRENT_TIMESTAMP(), '9999-12-31'::TIMESTAMP_LTZ, TRUE);

ALTER TASK VITECH_DEV.SILVER.ORDERS_TASK_SCD2 RESUME;


select * FROM VITECH_DEV.BRONZE.ORDERS_STREAM_SCD2;

select * from VITECH_DEV.SILVER.ORDERS_SCD2 ;

-- =============================================================
-- SCD TYPE 3: Partial history (stores previous value in columns)
-- =============================================================

CREATE OR REPLACE TABLE VITECH_DEV.SILVER.ORDERS_SCD3 (
    ORDER_ID NUMBER PRIMARY KEY,
    CUSTOMER_ID NUMBER NOT NULL,
    ORDER_DATE DATE NOT NULL,
    ORDER_STATUS VARCHAR(50) NOT NULL,
    PREV_ORDER_STATUS VARCHAR(50),
    TOTAL_AMOUNT NUMBER(12,2) NOT NULL,
    PREV_TOTAL_AMOUNT NUMBER(12,2),
    SHIPPING_ADDRESS VARCHAR(500),
    PREV_SHIPPING_ADDRESS VARCHAR(500),
    LAST_UPDATED TIMESTAMP_LTZ DEFAULT CURRENT_TIMESTAMP()
);

CREATE OR REPLACE STREAM VITECH_DEV.BRONZE.ORDERS_STREAM_SCD3
    ON TABLE VITECH_DEV.BRONZE.ORDERS APPEND_ONLY = FALSE;

CREATE OR REPLACE TASK VITECH_DEV.SILVER.ORDERS_TASK_SCD3
    WAREHOUSE = COMPUTE_WH
    SCHEDULE = 'USING CRON * * * * * UTC'
    WHEN SYSTEM$STREAM_HAS_DATA('VITECH_DEV.BRONZE.ORDERS_STREAM_SCD3')
    AS
MERGE INTO VITECH_DEV.SILVER.ORDERS_SCD3 AS TGT
USING VITECH_DEV.BRONZE.ORDERS_STREAM_SCD3 AS SRC
    ON TGT.ORDER_ID = SRC.ORDER_ID
WHEN MATCHED AND SRC.METADATA$ACTION = 'DELETE' AND SRC.METADATA$ISUPDATE = FALSE THEN
    DELETE
WHEN MATCHED AND SRC.METADATA$ACTION = 'INSERT' THEN
    UPDATE SET
        TGT.PREV_ORDER_STATUS = TGT.ORDER_STATUS,
        TGT.PREV_TOTAL_AMOUNT = TGT.TOTAL_AMOUNT,
        TGT.PREV_SHIPPING_ADDRESS = TGT.SHIPPING_ADDRESS,
        TGT.CUSTOMER_ID = SRC.CUSTOMER_ID,
        TGT.ORDER_DATE = SRC.ORDER_DATE,
        TGT.ORDER_STATUS = SRC.ORDER_STATUS,
        TGT.TOTAL_AMOUNT = SRC.TOTAL_AMOUNT,
        TGT.SHIPPING_ADDRESS = SRC.SHIPPING_ADDRESS,
        TGT.LAST_UPDATED = CURRENT_TIMESTAMP()
WHEN NOT MATCHED AND SRC.METADATA$ACTION = 'INSERT' THEN
    INSERT (ORDER_ID, CUSTOMER_ID, ORDER_DATE, ORDER_STATUS, TOTAL_AMOUNT, SHIPPING_ADDRESS, LAST_UPDATED)
    VALUES (SRC.ORDER_ID, SRC.CUSTOMER_ID, SRC.ORDER_DATE, SRC.ORDER_STATUS, SRC.TOTAL_AMOUNT, SRC.SHIPPING_ADDRESS, CURRENT_TIMESTAMP());

ALTER TASK VITECH_DEV.SILVER.ORDERS_TASK_SCD3 RESUME;


select * from VITECH_DEV.BRONZE.ORDERS_STREAM_SCD3 ;

select * from VITECH_DEV.SILVER.ORDERS_SCD3 ;
