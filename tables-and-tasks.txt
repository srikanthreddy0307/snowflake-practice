-- cron expression


--  __________ minute (0-59)
--  | ________ hour (0-23)
--  | | ______ day of month (1-31, or L)
--  | | | ____ month (1-12, JAN-DEC)
--  | | | | __ day of week (0-6, SUN-SAT, or L)
--  | | | | |
--  | | | | |
--  * * * * *

-- 0/5 * * * * UTC

-- Every five minutes, starting at the top of the hour. UTC time zone.

-- 5 * * * * UTC

-- The 5th minute of every hour. UTC time zone.

-- 30 3 * * * UTC

-- Every night at 3:30 a.m. UTC time zone.

-- 0 6,18 * * * UTC

-- Twice daily, at 6:00 a.m. and 6:00 p.m.UTC time zone.

-- 0 3 * * MON-FRI UTC

-- Weekdays at 3:00 a.m. UTC time zone.

-- 0 0 1 * * UTC

-- At midnight on the first day of every month. UTC time zone.

-- 0 0 L * * UTC

-- At midnight on the last day of every month. UTC time zone.

-- task using cron expressions
CREATE OR REPLACE TASK OUR_FIRST_DB.PUBLIC.NETFLIX_CRON_TASK
  WAREHOUSE = COMPUTE_WH
  SCHEDULE = 'USING CRON 0/5 * * * * UTC'
  AS
    INSERT OVERWRITE INTO OUR_FIRST_DB.PUBLIC.NETFILX_FINAL
    SELECT * FROM OUR_FIRST_DB.PUBLIC.NETFLIX_DATA_DYN;

ALTER TASK OUR_FIRST_DB.PUBLIC.NETFLIX_CRON_TASK RESUME;
